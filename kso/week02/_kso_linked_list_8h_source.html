5// 功能描述：单链表容器。提供基于迭代器的接口。
6//
7// 定义宏 KSO_SAFETY_CHECKED = 1 可以开启迭代器安全检查。开启后错误操作
8// 会导致程序中止，而不会造成UB。
9//
10// 原理：开启安全检查后，链表结点增加一个成员 m_lifeInd，类型为 shared_ptr。
11// 迭代器有它所对应链表结点的 m_lifeInd 的 weak_ptr 。
12// 在必要时，迭代器会检查 weak_ptr::expired() 的返回值
13// 来判断是否指向存活的链表结点。

231 Iterator insertAfter(ConstIterator iter, T &&obj)
232 {
233#if KSO_SAFETY_CHECKED
234 // 检查各种迭代器错误
235 // head (ok)
236 // tail (ok)
237 // null (x)
238 if (iter.m_iter.m_targetList != this)
239 {
240 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__NOT_MY_ITER);
241 }
242 if (iter.isEmpty())
243 {
244 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__NULL_ITER);
245 }
246 if (iter.isExpired())
247 {
248 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__ITER_EXPIRED);
249 }
250#endif
251 // 更新结点
252 iter.m_iter.m_node->m_next = new Node(std::move(obj), iter.m_iter.m_node->m_next);
253 // 维护 m_tail
254 if (iter.m_iter.m_node == m_tail)
255 {
256 m_tail = iter.m_iter.m_node->m_next;
257 }
258 // 维护 m_length
259 ++m_length;
260 // 返回的迭代器指向插入的结点
261 return Iterator(
262#if KSO_SAFETY_CHECKED
263 this,
264#endif
265 iter.m_iter.m_node->m_next);
266 }

280 Iterator eraseAfter(ConstIterator iter)
281 {
282#if KSO_SAFETY_CHECKED
283 // 检查各种迭代器错误
284 // head (ok)
285 // tail (x)
286 // null (x)
287 if (iter.m_iter.m_targetList != this)
288 {
289 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__NOT_MY_ITER);
290 }
291 if (iter.m_iter.isEmpty())
292 {
293 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__NULL_ITER);
294 }
295 if (iter.m_iter.isExpired())
296 {
297 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__ITER_EXPIRED);
298 }
299 if (iter.m_iter.m_node == m_tail)
300 {
301 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__OUT_OF_RANGE);
302 }
303#endif
304 // 更新结点
305 auto node_to_delete = iter.m_iter.m_node->m_next;
306 iter.m_iter.m_node->m_next = node_to_delete->m_next;
307 delete node_to_delete; // NOTE: 此后不要使用 node_to_delete !!
308 // 维护 m_tail
309 if (node_to_delete == m_tail)
310 {
311 m_tail = iter.m_iter.m_node;
312 }
313 // 维护 m_length
314 --m_length;
315 // 返回被删除结点的下一个结点
316 return Iterator(
317#if KSO_SAFETY_CHECKED
318 this,
319#endif
320 iter.m_iter.m_node->m_next);
321 }

449template <typename T>
450class LinkedList<T>::Iterator
451{
452 friend class LinkedList<T>;
453 friend class LinkedList<T>::ConstIterator;
454
455public:
456 using iterator_category = std::forward_iterator_tag;
457 using value_type = T;
458 using difference_type = std::ptrdiff_t;
459 using pointer = T *;
460 using reference = T &;
461
462private:
464 Node *m_node = nullptr;
465
466#if KSO_SAFETY_CHECKED
468 std::weak_ptr<LifeIndicator> m_lifeInd;
469
471 LinkedList<T> *m_targetList = nullptr;
472#endif
473
474public:
476 Iterator() = default;
477
478#if KSO_SAFETY_CHECKED
480 explicit Iterator(LinkedList<T> *targetList, Node *node)
481 {
482 Iterator::init(this, targetList, node);
483 }
484#else
486 explicit Iterator(Node *node)
487 {
488 Iterator::init(this, node);
489 }
490#endif
491
493 void operator++()
494 {
495#if KSO_SAFETY_CHECKED
496 // 空迭代器不可前进
497 if (this->isEmpty())
498 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__NULL_ITER);
499#endif
500 // 迭代器前进一步
501 m_node = m_node->m_next;
502
503#if KSO_SAFETY_CHECKED
504 if (!this->isEmpty())
505 // 如果前进后非空，则生命指示器也前进一步
506 m_lifeInd = m_node->m_lifeInd;
507 else
508 // 如果前进后变成空，则生命指示器取消引用
509 m_lifeInd.reset();
510#endif
511 }
512
515 T &operator*() const
516 {
517#if KSO_SAFETY_CHECKED
518 if (this->isEmpty()) // 检查迭代器是否空
519 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__NULL_ITER);
520 if (this->isExpired()) // 检查迭代器是否过期
521 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__ITER_EXPIRED);
522 if (this->m_node->isHead) // 不能解引用头结点
523 ABORT_WITH_MSG(KSO_LINKEDLIST_ERR__CANNOT_DEREF_HEAD);
524#endif
525 return m_node->m_data;
526 }
527
528 T *operator->() const
529 {
530 return &**this;
531 }
532
535 bool operator!=(Iterator const &other) const
536 {
537 return m_node != other.m_node;
538 }
539
542 bool operator==(Iterator const &other) const
543 {
544 return m_node == other.m_node;
545 }
546
548 bool isEmpty() const
549 {
550 return !m_node;
551 }
552
553#if KSO_SAFETY_CHECKED
555 bool isExpired() const
556 {
557 return m_lifeInd.expired();
558 }
559#endif
560
561private:
562#if KSO_SAFETY_CHECKED
565 static void init(Iterator *target, LinkedList<T> *targetList, Node *node)
566 {
567 target->m_targetList = targetList;
568 target->m_node = node;
569 if (node)
570 target->m_lifeInd = node->m_lifeInd;
571 }
572#else
575 static void init(Iterator *target, Node *node)
576 {
577 target->m_node = node;
578 }
579#endif
580};
